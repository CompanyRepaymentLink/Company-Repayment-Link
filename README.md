<script>

function generateLink(){
    let name = document.getElementById("name").value.trim();
    let upi = document.getElementById("upi").value.trim();
    let amount = document.getElementById("amount").value.trim();

    if(!name || !upi || !amount){
        alert("Please fill all fields");
        return;
    }

    let url = window.location.origin + window.location.pathname +
    "?name=" + encodeURIComponent(name) +
    "&upi=" + encodeURIComponent(upi) +
    "&amount=" + encodeURIComponent(amount);

    document.getElementById("result").innerHTML =
    "<p style='word-wrap:break-word;font-size:12px;'>" + url + "</p>";
}

function loadPayment(){
    const params = new URLSearchParams(window.location.search);

    if(params.has("name")){
        document.getElementById("generator").style.display="none";
        document.getElementById("viewer").style.display="block";

        document.getElementById("v_name").innerText=params.get("name");
        document.getElementById("v_upi").innerText=params.get("upi");
        document.getElementById("v_amount").innerText=params.get("amount")+" / ₹";
    }
}

function copyText(id){
    let text=document.getElementById(id).innerText;
    navigator.clipboard.writeText(text);
    alert("Copied: "+text);
}

function submitPayment(){
    let txn=document.getElementById("txn_input").value.trim();
    if(!txn){
        alert("Please enter UTR Number");
        return;
    }
    document.getElementById("successMsg").style.display="block";
}

loadPayment();
